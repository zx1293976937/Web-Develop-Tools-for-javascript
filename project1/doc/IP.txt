[title]
IP

[description]
验证DOM对象的value属性值, 确保属性值符合IP地址规则. 会在两种情况时触发验证, 一是DOM对象失去焦点时, 二是调用milk.validater.FormObject对象的validate函数时. 此函数实际为regExp函数的上层封装.

[dependencies]
jquery.milk-1.0.0.js
milk.validater-1.0.0.js

[returnType]
jQuery

[parameters]
String	errMsg	验证不通过时的通知消息

[example]
<!-- 示例一 -->
<input type="text" id="txt" />
<script>
	$("#txt").IP("IP地址不合法")
		.onError(function(sender, err) {
			alert(err);
			// 在输入非法IP时获得提示: IP地址不合法
		});
	// 注意! 需要提供onError处理函数才能获得通知消息
</script>
